# Vercel Postgres 数据库配置示例
# 从 Vercel Dashboard > Storage > 你的数据库 > .env.local 复制实际值

POSTGRES_PRISMA_URL="postgresql://username:password@ep-xxx.us-east-1.postgres.vercel-storage.com:5432/verceldb?pgbouncer=true&connect_timeout=15"
POSTGRES_URL_NON_POOLING="postgresql://username:password@ep-xxx.us-east-1.postgres.vercel-storage.com:5432/verceldb"
